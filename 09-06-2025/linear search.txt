public class Main {
    public static void main(String[] args) {
        int[] nums = {10, 19 ,21, 2, 3, 1, 8, 9};
        int target=9;
        int index=-1;
        for(int i=0;i<nums.length;i++){
            if(target==nums[i]);
            index=i;
            break;
        }
        if (index == -1) {
            System.out.println("Element not found");
        } else {
            System.out.println("Element found at index: " + index);
    }
        
    }
}
